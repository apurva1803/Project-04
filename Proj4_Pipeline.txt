pipeline {
    agent any

    environment {
        JAVA_HOME = "/opt/homebrew/Cellar/openjdk@17/17.0.18/libexec/openjdk.jdk/Contents/Home"
        MAVEN_HOME = "/opt/homebrew/opt/maven"

        TOMCAT_HOME = "/opt/homebrew/opt/tomcat@9/libexec"
        TOMCAT_WEBAPPS_PATH = "${TOMCAT_HOME}/webapps"

        PATH = "/opt/homebrew/bin:/usr/local/bin:${JAVA_HOME}/bin:${MAVEN_HOME}/bin:${PATH}"
    }

    stages {

        stage('Cleanup Workspace') {
            steps {
                deleteDir()
            }
        }

        // ✅ OPTIONAL: Start only if not running
        stage('Ensure Tomcat Running') {
            steps {
                sh '''
                echo "Checking if Tomcat is running on port 1999..."

                if lsof -i :1999 > /dev/null
                then
                    echo "Tomcat already running"
                else
                    echo "Starting Tomcat..."
                    brew services start tomcat@9
                    echo "Waiting for Tomcat startup..."
                    sleep 20
                fi
                '''
            }
        }

        stage('Clone Repository') {
            steps {
                git url: 'https://github.com/apurva1803/Project-04.git',
                    branch: 'main'
            }
        }

        stage('Build WAR File') {
            steps {
                sh "mvn clean package -DskipTests -f ORSProject-04/pom.xml"
            }
        }

        stage('Deploy WAR to Tomcat') {
            steps {
                script {
                    def warFile = "ORSProject-04/target/ORSProject-04.war"
                    def destination = "${TOMCAT_WEBAPPS_PATH}/ORSProject-04.war"

                    echo "Removing old deployment..."
                    sh "rm -rf ${TOMCAT_WEBAPPS_PATH}/ORSProject-04"
                    sh "rm -f ${destination}"

                    echo "Deploying new WAR..."
                    sh "cp ${warFile} ${destination}"

                    echo "Waiting for WAR extraction..."
                    sleep 15
                }
            }
        }

        // ✅ FIXED: Stateless health check (NO SESSION)
        stage('Verify Deployment (Robust)') {
            steps {
                sh '''
                echo "Checking application health..."

                for i in {1..15}
                do
                  STATUS=$(curl -I -s -o /dev/null -w "%{http_code}" http://localhost:1999/ORSProject-04/)

                  if [ "$STATUS" = "200" ] || [ "$STATUS" = "302" ]; then
                    echo "✅ Application is UP (HTTP $STATUS)"
                    exit 0
                  fi

                  echo "Retry $i - Status: $STATUS"
                  sleep 5
                done

                echo "❌ Application failed to start"
                exit 1
                '''
            }
        }
    }

    post {
        always {
            deleteDir()
        }
    }
}
